    public void fetchAllMealTest(DatabaseReference mealDb) {
        List<Meal> mealList = new ArrayList<Meal>();
        mealDb.addValueEventListener(new ValueEventListener() {
            @Override
            public void onDataChange(@NonNull DataSnapshot snapshot) {

                for (DataSnapshot mealSnapshot: snapshot.getChildren()) {
                    meal = mealSnapshot.getValue(Meal.class);
                    mealList.add(meal);
                }
                //Test
                // System.out.println(Collections.singletonList(mealList.get(1)));
                System.out.println(Collections.singletonList(mealList.get(2)));
            }

            @Override
            public void onCancelled(@NonNull DatabaseError error) {
                Log.w("TAG", "loadMeal :onCancelled", error.toException());
            }
        });
    }
